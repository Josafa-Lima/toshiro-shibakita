version: '3.8'

services:

  # Banco de dados MySQL
  mysql:
    image: mysql:5.7
    container_name: mysql-db
    environment:
      MYSQL_ROOT_PASSWORD: Senha123
      MYSQL_DATABASE: meubanco
    volumes:
      - ./banco.sql:/docker-entrypoint-initdb.d/banco.sql
    networks:
      - app-network

  # Instancia 1 da aplicacao PHP
  php-app1:
    build:
      context: .
      dockerfile: Dockerfile-php
    environment:
      MYSQL_HOST: mysql
      MYSQL_USER: root
      MYSQL_PASS: Senha123
      MYSQL_DB: meubanco
    depends_on:
      - mysql
    networks:
      - app-network

  # Instancia 2 da aplicacao PHP
  php-app2:
    build:
      context: .
      dockerfile: Dockerfile-php
    environment:
      MYSQL_HOST: mysql
      MYSQL_USER: root
      MYSQL_PASS: Senha123
      MYSQL_DB: meubanco
    depends_on:
      - mysql
    networks:
      - app-network

  # Instancia 3 da aplicacao PHP
  php-app3:
    build:
      context: .
      dockerfile: Dockerfile-php
    environment:
      MYSQL_HOST: mysql
      MYSQL_USER: root
      MYSQL_PASS: Senha123
      MYSQL_DB: meubanco
    depends_on:
      - mysql
    networks:
      - app-network

  # Load Balancer NGINX
  nginx:
    build: .
    ports:
      - '4500:4500'
    depends_on:
      - php-app1
      - php-app2
      - php-app3
    networks:
      - app-network

networks:
  app-network:
    driver: bridge
